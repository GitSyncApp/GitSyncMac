import Foundation


struct TaskItem {
    var bool: String
    var id: String
}

var customArray = [
    TaskItem(bool: "X", id:"1"),
    TaskItem(bool: " ", id:"2"),
    TaskItem(bool: "X", id:"3"),
    TaskItem(bool: " ", id:"4"),
    TaskItem(bool: " ", id:"5")
]

/**
 *
 */
func sort(arr:Array<TaskItem>, _ reverse:Bool = false)->Array<TaskItem>{
    var completeTasks:Array<TaskItem> = []
    var inCompleteTasks:Array<TaskItem> = []
    for taskItem in arr{
        (taskItem.bool == "X") != reverse ? completeTasks.append(taskItem) : inCompleteTasks.append(taskItem)
    }
   let returnArr = completeTasks + inCompleteTasks
   return returnArr
}
sort(customArray,false)
